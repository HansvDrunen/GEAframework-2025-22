# --- 4) Final cleaner (language-agnostic; works for mixed EN/NL)
token_pat = re.compile(r'[^a-zA-Z]+')

def clean_text(text: str) -> str:
    text = str(text).lower()
    text = token_pat.sub(' ', text)
    toks = [w for w in text.split()
            if len(w) > 2 and w not in stop_words and not w.isdigit()]
    # drop ultra-rare tokens learned above
    toks = [w for w in toks if w not in very_rare]
    return " ".join(toks)

# Plot Frequent words
def plot_word_freq(data_subset, title, token_col="tokens_clean", topn=15):
    # pull flattened token list
    all_tokens = [t for toks in data_subset[token_col].dropna() for t in toks]
    if not all_tokens:
        print(f"No tokens to plot for {title}")
        return
    freq = Counter(all_tokens).most_common(topn)
    words, counts = zip(*freq)

    plt.figure(figsize=(10,5), dpi=130)
    plt.bar(words, counts, edgecolor="black")
    plt.xticks(rotation=45, ha="right")
    plt.title(title)
    plt.tight_layout()
    plt.show()

KMeans Clustering per language per industry

This cell performs automatic clustering for each language and industry subset using K-Means with an optimized number of clusters based on the cosine silhouette score. Instead of fixing the number of clusters, the algorithm tests several values and selects the one that best fits the data. It then calculates cluster purity and silhouette scores to evaluate how consistent and well-separated the clusters are. The printed results show, for each subset, the chosen number of clusters, the achieved purity, and the silhouette score, which together indicate the overall quality of the clustering.

# Purity metric
def cluster_purity(y_true, y_pred):
    """
    Purity = sum of the majority class counts per cluster / total samples.
    Compares predicted clusters against your ground-truth 'Type' labels.
    """
    ct = pd.crosstab(pd.Series(y_true, name="true"),
                     pd.Series(y_pred, name="cluster"))
    return (ct.max(axis=0).sum() / ct.values.sum()) if ct.size else np.nan

# Pick k automatically via silhouette (cosine)
def pick_k_by_silhouette(X, k_min=2, k_max=10, random_state=42):
    """
    Try k in [k_min, k_max], pick the one with the best cosine silhouette.
    Skips invalid k for tiny samples. Returns (best_k, best_labels, best_model, scores_dict).
    """
    n = X.shape[0]
    usable_max = max(k_min, min(k_max, n - 1))  # need at least k+1 samples
    scores = {}
    best = {"k": None, "score": -1, "labels": None, "model": None}

    for k in range(k_min, usable_max + 1):
        km = KMeans(n_clusters=k, n_init="auto", random_state=random_state)
        labels = km.fit_predict(X)

        # silhouette needs at least 2 different labels and < n unique labels
        if len(set(labels)) < 2 or len(set(labels)) >= n:
            continue

        try:
            s = silhouette_score(X, labels, metric="cosine")
        except Exception:
            continue

        scores[k] = s
        if s > best["score"]:
            best.update({"k": k, "score": s, "labels": labels, "model": km})

    # Fallback if nothing valid (rare): force k=2
    if best["k"] is None:
        k = min(2, max(2, usable_max))
        km = KMeans(n_clusters=k, n_init="auto", random_state=random_state)
        labels = km.fit_predict(X)
        best = {"k": k, "score": np.nan, "labels": labels, "model": km}
        scores[k] = np.nan

    return best["k"], best["labels"], best["model"], scores

# Cluster themes
bad = {"mission","vision","strategy","goals","core value","company","corporation",
       "create","drive","growth","commitment","experience","step ahead","stay step"}

def top_phrases_distinct(texts, labels, topn=8):
    clusters = sorted(set(labels))
    docs = [" ".join([texts[i] for i in range(len(texts)) if labels[i]==c]) for c in clusters]

    def _fit_counts(ngram_range):
        cv = CountVectorizer(ngram_range=ngram_range, min_df=1)  # <- changed to 1
        Xc = cv.fit_transform(docs)
        return cv, Xc

    # try bigrams first; if empty, fall back to (1,2)
    try:
        cv, Xc = _fit_counts((2,2))
        if Xc.shape[1] == 0:
            raise ValueError
    except Exception:
        cv, Xc = _fit_counts((1,2))

    terms = np.array(cv.get_feature_names_out())

    # c-TF-IDF scoring
    tf = Xc.astype(float)
    tf = tf / (tf.sum(axis=1) + 1e-12)
    df_terms = (Xc > 0).sum(axis=0).A1
    idf = np.log((len(clusters)+1) / (df_terms+1)) + 1.0
    scores = tf.multiply(idf)

    def norm(t): return re.sub(r"[^a-z0-9\s\-]+","",t.lower()).strip()

    out = {}
    for i,c in enumerate(clusters):
        row = np.asarray(scores.getrow(i).todense()).ravel()
        idx = row.argsort()[::-1]
        picks, seen = [], []
        for j in idx:
            ph = terms[j]; phn = norm(ph)
            if phn in bad or len(phn) < 3: 
                continue
            toks = set(phn.split())
            if any(len(toks & set(s.split()))/len(toks | set(s.split())) > 0.6 for s in seen):
                continue
            seen.append(phn); picks.append(ph)
            if len(picks) == topn: break
        out[c] = picks
    return out

MissionGoals Semantic Alignment by Industry

We will create a prototype that measures how closely each companys mission and goals statements align in meaning. Using TF-IDF, well convert both texts into numerical vectors and calculate the cosine similarity between them. This produces a coherence score for each company, showing how semantically consistent their mission and goals are.

In[22]:
# Build simple per-company mission/goals strings
mg = (data
      .pivot_table(index=["Company","Industry","lang"],
                   columns="Type",
                   values="clean",
                   aggfunc=lambda x: " ".join(x))
      .reset_index())

# keep rows where both exist
mg = mg[mg["mission"].notna() & mg["goals"].notna()].copy()

# Vectorize both columns jointly to a common space
vec = TfidfVectorizer(min_df=1, ngram_range=(1,2), norm="l2")
M = vec.fit_transform(mg["mission"].fillna(""))
G = vec.transform(mg["goals"].fillna(""))

# Cosine similarity
from sklearn.metrics.pairwise import cosine_similarity
mg["MissionGoalsCosine"] = np.diag(cosine_similarity(M, G))

# Company-level report (optional)
mg[["Company","Industry","lang","MissionGoalsCosine"]].sort_values(
    "MissionGoalsCosine", ascending=False).head(10)

The prototype shows how tightly each industry aligns its mission with its goals using TF-IDF cosine similarity: Real Estate stands out with the highest coherence (0.69), followed by Healthcare (0.49), Consumer Services (0.44) and Utilities (0.41), suggesting these sectors articulate goals that closely echo their missions. In contrast, Technology (0.20), Public Sector (0.18), Transport (0.17) and Finance (0.15) are more diffusemissions and goals share less wording or emphasize different ideas; Retail and Hospitality appear near zero, likely because few firms had both mission and goals in the current sample. At the company level this pattern is visible too: e.g., De Key (Real Estate, nl) scores 0.69, Lufthansa (Transport, en) ~0.51, Philips (Healthcare, en) 0.49, while IBM, Brabant Water, Ordina, AMD, and Kadaster sit in the mid-range (0.270.44). Treat these as directional insights rather than ground truth: the metric captures textual similarity (not strategic success), and results can shift with dataset size, language mix, and how statements are written.
